#include <stdio.h>

int main()
{
    char blood;//한글자
    printf("혈액형을 입력(A,B,C):");
    scanf("%c", &blood);
    
    //조건문 witch ~case
    switch(blood)//case는 두줄을 넣어도 중괄호를 쓰지않음.
    {
        case 'A'://case뒤에 A는 하나의 문자이기 때문에 작은따옴표''를쓴다.
            printf("A형입니다.\n");
            break;
        case 'B':
            printf("B형입니다.\n");
            break;
        case 'O':
            printf("O형입니다.\n");
            break;
        default:
        printf("입력오류.");
        
        
    }
    return 0;
}
